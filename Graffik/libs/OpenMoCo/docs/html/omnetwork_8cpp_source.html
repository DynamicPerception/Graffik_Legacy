634 void OMNetwork::_complete(OMCommandBuffer *buf) {
635 
636  // if a broadcast command, don't store in history,
637  // but do reflect signal
638  if( buf->broadcast() ) {
639  emit complete(buf);
640  return;
641  }
642 
643  // odd, we don't have any such device known to us...
644 
645  if( ! m_devIds.contains(buf->device()) )
646  return;
647 
648  OMdeviceInfo* dev = m_devIds[ buf->device() ];
649 
650  if( buf->status() == OMC_FAILURE )
651  dev->errors++;
652 
653  if( m_histCnt > 0 ) {
654  // need to check and make sure that we haven't exceeded
655  // the requested history count, if so - purge any
656  // command but this one from the history
657  while( dev->commandHistory->count() > m_histCnt ) {
658  int cmdId = dev->commandHistory->keys()[0];
659  if( cmdId == buf->id() )
660  continue;
661  delete dev->commandHistory->value(cmdId);
662  dev->commandHistory->remove(cmdId);
663  }
664  }
665 
666  emit complete(buf->id(), buf);
667  emit complete(buf);
668 
669 }
